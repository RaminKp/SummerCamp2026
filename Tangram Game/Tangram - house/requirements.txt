# Tangram Quest - House (Misty II)  -  Python dependencies
# Install with:  pip install -r requirements.txt
#
# IMPORTANT: this game uses cv2.aruco, which ships ONLY in the *contrib*
# build of OpenCV. Do NOT also install plain "opencv-python" in the same
# environment - the two packages clash. If you already have opencv-python,
# run:  pip uninstall opencv-python  before installing the line below.

opencv-contrib-python>=4.7.0      # ArUco detection + Haar face cascade (tested on 4.13)
numpy>=1.21                       # array math (tested on 2.4)
requests>=2.25                    # talking to Misty II over REST (tested on 2.33)

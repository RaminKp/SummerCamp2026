# ── Blossom Maze Game — Python Dependencies ────────────────────────────────────
# Install with: pip install -r requirements.txt
#
# Note: RPi.GPIO and mfrc522 are Raspberry Pi specific.
#       Run pip install on the Pi itself, not on a dev laptop.
# ──────────────────────────────────────────────────────────────────────────────

# HTTP / REST — misty.py, narrator.py, head_track.py, movement_test.py
requests

# WebSocket client — head_track.py (import websocket)
websocket-client

# RFID hardware — rfid_test.py, detector.py, enrol_cards.py
# rpi-lgpio is a drop-in replacement for RPi.GPIO (same "RPi.GPIO" import name)
# that works on the Pi 5's BCM2712 SoC, which RPi.GPIO itself does not support.
# Do not install both — they conflict.
rpi-lgpio
mfrc522

# Computer vision + ArUco (used in misty.recalibrate_at_home)
# opencv-contrib-python includes the aruco module; use the headless variant
# on the Pi to avoid pulling in a GUI display stack.
opencv-contrib-python-headless
numpy

# Excel game logging — game_logger.py
openpyxl
